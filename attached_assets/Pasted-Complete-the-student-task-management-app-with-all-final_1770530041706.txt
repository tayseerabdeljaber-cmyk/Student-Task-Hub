```typescript
interface Recommendation {
  id: string;
  type: 'start_early' | 'balance_workload' | 'exam_prep' | 'difficult_assignment' | 'office_hours' | 'break_needed';
  priority: 'low' | 'medium' | 'high';
  title: string;
  description: string;
  action?: {
    label: string;
    handler: () => void;
  };
  dismissible: boolean;
}

// Generate recommendations based on data
function generateRecommendations(
  assignments: Assignment[],
  schedule: ScheduleBlock[],
  userActivity: UserActivity
): Recommendation[] {
  const recommendations: Recommendation[] = [];
  
  // 1. Start early on big assignments
  assignments
    .filter(a => a.estimatedHours >= 4 && daysUntil(a.dueDate) >= 5)
    .forEach(assignment => {
      recommendations.push({
        type: 'start_early',
        priority: 'high',
        title: `Start ${assignment.title} early`,
        description: `This assignment typically takes ${assignment.estimatedHours}+ hours. Start now to avoid last-minute stress.`,
        action: {
          label: 'Schedule now',
          handler: () => scheduleAssignment(assignment.id)
        }
      });
    });
  
  // 2. Exam preparation
  assignments
    .filter(a => a.type === 'exam' && daysUntil(a.dueDate) <= 5)
    .forEach(exam => {
      const studyHoursScheduled = schedule.filter(
        b => b.assignmentId === exam.id
      ).reduce((sum, b) => sum + b.duration, 0);
      
      if (studyHoursScheduled < 4) {
        recommendations.push({
          type: 'exam_prep',
          priority: 'high',
          title: `${exam.courseName} exam in ${daysUntil(exam.dueDate)} days`,
          description: `You've only scheduled ${studyHoursScheduled} hours. Consider adding 2-3 more study sessions.`,
          action: {
            label: 'Add study time',
            handler: () => scheduleExamPrep(exam.id)
          }
        });
      }
    });
  
  // 3. Difficult assignments
  assignments
    .filter(a => a.difficulty === 'hard' && !a.isCompleted)
    .forEach(assignment => {
      recommendations.push({
        type: 'difficult_assignment',
        priority: 'medium',
        title: `${assignment.title} is challenging`,
        description: `Students typically struggle with this. Consider attending office hours or finding a study group.`,
        action: {
          label: 'View resources',
          handler: () => showResources(assignment.courseId)
        }
      });
    });
  
  // 4. Balance workload
  const weekSchedule = getThisWeekSchedule(schedule);
  const overloadedDays = weekSchedule.filter(day => day.totalHours > 10);
  const lightDays = weekSchedule.filter(day => day.totalHours < 4);
  
  if (overloadedDays.length > 0 && lightDays.length > 0) {
    recommendations.push({
      type: 'balance_workload',
      priority: 'medium',
      title: 'Unbalanced schedule this week',
      description: `${overloadedDays[0].name} has ${overloadedDays[0].totalHours} hours scheduled. Move some to ${lightDays[0].name}?`,
      action: {
        label: 'Rebalance',
        handler: () => rebalanceWeek()
      }
    });
  }
  
  // 5. Break needed
  const recentStudyHours = userActivity.last7Days.studyHours;
  if (recentStudyHours > 40) {
    recommendations.push({
      type: 'break_needed',
      priority: 'medium',
      title: 'You\'ve been working hard',
      description: `You've studied ${recentStudyHours} hours this week. Consider taking a break to avoid burnout.`,
      action: {
        label: 'Schedule free time',
        handler: () => scheduleFreeTime()
      }
    });
  }
  
  // 6. Office hours
  const strugglingCourses = userActivity.courses.filter(
    c => c.averageGrade < 75 || c.missedAssignments > 2
  );
  strugglingCourses.forEach(course => {
    recommendations.push({
      type: 'office_hours',
      priority: 'high',
      title: `Struggling in ${course.name}?`,
      description: `You've missed ${course.missedAssignments} assignments. Office hours: ${course.officeHours}`,
      action: {
        label: 'Add to calendar',
        handler: () => addOfficeHours(course.id)
      }
    });
  });
  
  return recommendations.sort((a, b) => {
    const priorityOrder = { high: 3, medium: 2, low: 1 };
    return priorityOrder[b.priority] - priorityOrder[a.priority];
  });
}
```

```typescript
function personalize Recommendations(
  baseRecommendations: Recommendation[],
  userActivity: UserActivity
): Recommendation[] {
  // Filter out recommendations for times user rarely completes
  return baseRecommendations.map(rec => {
    if (rec.type === 'start_early') {
      // Suggest user's most productive time
      const bestHour = Object.entries(userActivity.completionRate.byTimeOfDay)
```

```typescript
  const insights = [];
```
